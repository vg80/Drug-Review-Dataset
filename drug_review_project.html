Now lets discuss reviews by time. First format the date and then extract year month and day fromit.

Bydate = train%>%mutate(date = as.Date(date,format="%d-%B-%y"))%>% mutate(month=as.numeric(format(date,'%m')), year= as.numeric(format(date,'%Y')), day= as.numeric(format(date,'%d')), weekday=weekdays(date, abbreviate= T))

Number of Reviews as per year

Bydate%>% group_by(year)%>%summarise(no_of_reviews= n())%>% ggplot()+geom_line(aes(x= year, y= no_of_reviews), color="red")+labs(x=NULL, y= NULL) + ggtitle("Number of reviews per Year")+ scale_x_continuous(breaks = seq(2008,2017,1))+ scale_y_continuous(breaks = seq(0,30000,5000))+ theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank(),panel.background = element_blank(), axis.line = element_blank())

Bydate %>% group_by(month) %>% summarise(no_of_reviews= n())%>% ggplot() + geom_line(aes(x= month, y= no_of_reviews), color="red") + labs(x=NULL, y= NULL) + ggtitle("Number of reviews per Month") + theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank(),panel.background = element_blank(), axis.line = element_blank()) + scale_x_continuous(breaks = seq(1,12,1), labels = month.abb)+ scale_y_continuous(breaks = seq(500,15000,500))
